pub fn cors() -> Builder

Create a wrapping filter that exposes CORS behavior for a wrapped filter.

Example

use warp::Filter;

let cors = warp::cors()
    .allow_origin("https://hyper.rs")
    .allow_methods(vec!["GET", "POST", "DELETE"]);

let route = warp::any()
    .map(warp::reply)
    .with(cors);

If you want to allow any route:

use warp::Filter;
let cors = warp::cors()
    .allow_any_origin();
